Factory Pipeline: Infrastructure Setup

This script initializes the PostgreSQL environment. It creates the metadata registry (the "Work Orders") and the raw data landing zone (the "Storage Bin").

File 1: setup_db.py

Database Initialization Script

Run this first to build the factory floor.


import psycopg2
from psycopg2 import extras

# CONFIGURATION
DB_CONFIG = {
    "host": "localhost",
    "database": "factory_db",
    "user": "postgres",
    "password": "yourpassword"
}

def setup_infrastructure():
    conn = psycopg2.connect(**DB_CONFIG)
    cur = conn.cursor()

    print("--- 🏗️ Building Factory Infrastructure ---")

    # 1. CREATE METADATA TABLE (The Priority Queue Registry)
    cur.execute("""
        CREATE TABLE IF NOT EXISTS pipeline_metadata (
            table_id SERIAL PRIMARY KEY,
            schema_name TEXT NOT NULL,
            table_name TEXT NOT NULL,
            poll_interval_sec INTEGER DEFAULT 30,
            last_id BIGINT DEFAULT 0,
            last_timestamp TIMESTAMP DEFAULT '1970-01-01',
            is_active BOOLEAN DEFAULT TRUE,
            UNIQUE(schema_name, table_name)
        );
    """)

    # 2. CREATE RAW DATA LANDING ZONE (The Finished Goods Storage)
    # We use JSONB for flexibility, or you can map columns specifically.
    cur.execute("""
        CREATE TABLE IF NOT EXISTS raw_data_storage (
            uuid UUID PRIMARY KEY DEFAULT gen_random_uuid(),
            source_table TEXT,
            data JSONB,
            ingested_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
        );
    """)

    # 3. SEED INITIAL TABLES (Example Tables)
    seed_data = [
        ('AC', 'O_Production', 10),
        ('AC', 'I_Material', 30),
        ('SMARTMIS', 'Mould', 60),
        ('SmartIIOT', 'Equipment', 15)
    ]
    
    insert_query = """
        INSERT INTO pipeline_metadata (schema_name, table_name, poll_interval_sec)
        VALUES (%s, %s, %s)
        ON CONFLICT DO NOTHING;
    """
    cur.executemany(insert_query, seed_data)

    conn.commit()
    cur.close()
    conn.close()
    print("--- ✅ Factory Floor Ready ---")

if __name__ == "__main__":
    setup_infrastructure()
