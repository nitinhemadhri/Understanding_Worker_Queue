The Alarm & The Workers

This script manages the Priority Queue. It sleeps until the next table is due, then hands the job to one of the 8 workers.


import queue
import time
import concurrent.futures
from heapq import heappush, heappop

# The Priority Queue (Alarm Clock)
task_heap = [] 

def worker_task(table_info):
    # 1. Connect to MES (ODBC)
    # 2. Check for iD > last_id
    # 3. If rows found -> push to data_buffer
    print(f"Polling {table_info['name']}...")
    return True

# 8 People in the factory
with concurrent.futures.ThreadPoolExecutor(max_workers=8) as executor:
    while True:
        # Get next due table
        due_time, table = heappop(task_heap)
        
        # Wait until it's time
        now = time.time()
        if due_time > now:
            time.sleep(due_time - now)
            
        # Send to one of the 8 workers
        executor.submit(worker_task, table)
